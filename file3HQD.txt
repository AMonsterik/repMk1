#include <iostream>
#include <fstream>
#include <string>
#include <array>

using namespace std;

// Функция для создания файла и записи названий предметов
void createFile(const string& day, const array<string, 5>& subjects) {
    ofstream file(day + ".txt"); // Используем ofstream вместо fstream, если только запись
    if (!file) { // Проверяем, был ли файл открыт успешно
        cerr << "Ошибка при открытии файла: " << day << ".txt" << endl;
        return;
    }
    
    for (const auto& subject : subjects) {
        file << subject << endl; // Записываем каждый предмет на новой строке
    }
}

int main() {
    // Шаблоны дней недели
    const array<string, 7> days = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};
    // Шаблоны предметов
    const array<string, 5> subjects = {"Math", "Physics", "Chemistry", "Biology", "History"};

    // Создание текстовых файлов для каждого дня недели
    for (const auto& day : days) {
        createFile(day, subjects);
    }

    // Открываем созданные файлы в текстовом редакторе
    for (const auto& day : days) {
        string command = day + ".txt"; // Формируем команду для открытия файла
        system(command.c_str()); // Выполняем команду, открывая файл
    }

    return 0; // Завершаем программу
}
