#include <iostream>
#include <fstream>

using namespace std;

int main() {
    const string days[] = { "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday" };
    const string subjects[] = { "Math", "Physics", "Chemistry", "Biology", "History" };

    ofstream scheduleFile("schedule.json");
    if (!scheduleFile) return cerr << "Error: Unable to open output file." << endl, 1;

    scheduleFile << "{\n";
    for (const auto& day : days) {
        scheduleFile << " "" << day << "": [";
        for (int j = 0; j < 5; ++j) {
            scheduleFile << (j ? ", " : "") << """ << subjects[j] << """;
        }
        scheduleFile << "]\n";
    }
    scheduleFile << "}\n";

    cout << "Schedule written to 'schedule.json'" << endl;
}
